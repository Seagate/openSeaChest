#See https://mesonbuild.com/Cross-compilation.html
#and https://mesonbuild.com/Machine-files.html
#and https://mesonbuild.com/Reference-tables.html#cpu-families
#NOTE: This may work with debian as well.
#example usage of this file:
#    meson --buildtype=release --cross-file=./meson_crosscompile/powerpcle-linux-musl-cross.txt crossbuild_powerpclemusl
#    ninja -C crossbuild_powerpclemusl

#NOTE: This script requires the musl cross compiler to be installed in /opt/muslcc
#      the install_muslcc.sh script in this same directory will do this for you!


[binaries]
c = '/opt/muslcc/powerpcle-linux-musl-cross/usr/bin/powerpcle-linux-musl-gcc'
ar = '/opt/muslcc/powerpcle-linux-musl-cross/usr/bin/powerpcle-linux-musl-gcc-ar'
strip = '/opt/muslcc/powerpcle-linux-musl-cross/usr/bin/powerpcle-linux-musl-strip'
pkgconfig = '/opt/muslcc/powerpcle-linux-musl-cross/usr/bin/powerpcle-linux-musl-pkg-config'
ld = '/opt/muslcc/powerpcle-linux-musl-cross/usr/bin/powerpcle-linux-musl-ld'

[properties]
root = '/opt/muslcc/powerpcle-linux-musl-cross/usr/powerpcle-linux-musl'
has_function_printf = true
skip_sanity_check = true

[host_machine]
system = 'linux'
cpu_family = 'ppc'
cpu = 'powerpcle'
endian = 'little'

[built-in options]
#since this is the musl cross-compiler, we want to statically link the executable
#This is MIT licensed and compatible to do. It will also make this super portable!
#
default_library='static'
c_link_args='-static'
libc_musl=true
